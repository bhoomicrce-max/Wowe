      "title": "New Event",
      "category": "workshop",
      "description": "...",
      "location": "...",
      "hostId": 1,
      "price": 25
    }'

🔄 HOW TO USE WITH FRONTEND
════════════════════════════════════════════════════════════════════════════════

1. The FRONTEND-INTEGRATION.js file has ready-to-use functions:

   import { 
     getAllEvents,
     getEventsByCategory,
     searchEvents,
     bookEvent,
     getHostProfile
   } from './FRONTEND-INTEGRATION.js';

2. Example in React:
   
   function EventsList() {
     const [events, setEvents] = useState([]);
     
     useEffect(() => {
       getAllEvents(20, 0).then(res => setEvents(res.data));
     }, []);
     
     return events.map(e => <EventCard key={e.id} event={e} />);
   }

3. Vanilla JavaScript integration already works - just update API calls
   in existing frontend code to use the backend endpoints
